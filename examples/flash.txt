int32 DELAY = 300;
int32 BASEPIN = 2;

for(int32 j = BASEPIN; j < BASEPIN + 5; j = j + 1;) {
	exportpin(j);
	directpin(j, 1);
}

for(int32 i = 0; 1; i = 1 - i;) {
	for(j = 0; j < 5; j = j + 1;) {
		writepin(BASEPIN + j, i);
	}
	sleep(DELAY);
}

for(j = BASEPIN; j < BASEPIN + 5; j = j + 1;) {
	unexportpin(j);
}
